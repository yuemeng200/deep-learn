```go
package models

import (
    "errors"

    "gorm.io/gorm"
)

type Task struct {
    gorm.Model
    Title     string `gorm:"size:200;not null"`
    Status    string `gorm:"size:20;default:'todo'"`
    Priority  int    `gorm:"default:0"`
    CreatorID uint
    Creator   User
    Tags      []Tag `gorm:"many2many:task_tags;"`
    Urgent    bool  `gorm:"-"`
}

func (t *Task) BeforeCreate(tx *gorm.DB) error {
    if t.Title == "" {
        return errors.New("title is required")
    }
    if t.Status == "" {
        t.Status = "todo"
    }
    return nil
}

func (t *Task) AfterFind(tx *gorm.DB) error {
    t.Urgent = t.Priority >= 8
    return nil
}
```

## 5.3 Repository 层

```go
// repository/user_repo.go
package repository

import (
    "errors"
    "taskflow/database"
    "taskflow/models"

    "gorm.io/gorm"
)

type UserRepo struct{}

func (r *UserRepo) Create(user *models.User) error {
    return database.DB.Create(user).Error
}

func (r *UserRepo) GetByID(id uint) (*models.User, error) {
    var user models.User
    err := database.DB.Preload("Profile").First(&user, id).Error
    if errors.Is(err, gorm.ErrRecordNotFound) {
        return nil, nil
    }
    return &user, err
}

func (r *UserRepo) List(page, pageSize int) ([]models.User, int64, error) {
    var users []models.User
    var total int64

    database.DB.Model(&models.User{}).Count(&total)
    err := database.DB.
        Offset((page - 1) * pageSize).
        Limit(pageSize).
        Find(&users).Error
    return users, total, err
}

func (r *UserRepo) Update(user *models.User) error {
    return database.DB.Model(user).Updates(map[string]interface{}{
        "Name":  user.Name,
        "Email": user.Email,
    }).Error
}
```

```go
// repository/task_repo.go
package repository

import (
    "errors"
    "taskflow/database"
    "taskflow/models"

    "gorm.io/gorm"
)

type TaskRepo struct{}

func (r *TaskRepo) Create(task *models.Task, tags []models.Tag) error {
    return database.DB.Transaction(func(tx *gorm.DB) error {
        if err := tx.Create(task).Error; err != nil {
            return err
        }
        if len(tags) > 0 {
            if err := tx.Model(task).Association("Tags").Append(tags); err != nil {
                return err
            }
        }
        return nil
    })
}

func (r *TaskRepo) GetByID(id uint) (*models.Task, error) {
    var task models.Task
    err := database.DB.Preload("Creator").Preload("Tags").
        First(&task, id).Error
    if errors.Is(err, gorm.ErrRecordNotFound) {
        return nil, nil
    }
    return &task, err
}

func (r *TaskRepo) List(page, pageSize int, keyword string) ([]models.Task, int64, error) {
    var tasks []models.Task
    var total int64

    query := database.DB.Model(&models.Task{})
    if keyword != "" {
        query = query.Where("title LIKE ?", "%"+keyword+"%")
    }

    query.Count(&total)
    err := query.Preload("Creator").Preload("Tags").
        Order("priority desc, created_at desc").
        Offset((page - 1) * pageSize).
        Limit(pageSize).
        Find(&tasks).Error
    return tasks, total, err
}

func (r *TaskRepo) Assign(taskID, userID uint) error {
    return database.DB.Transaction(func(tx *gorm.DB) error {
        if err := tx.Model(&models.Task{}).Where("id = ?", taskID).
            Update("assignee_id", userID).Error; err != nil {
            return err
        }
        log := models.OperationLog{TaskID: taskID, UserID: userID, Action: "assigned"}
        return tx.Create(&log).Error
    })
}

func (r *TaskRepo) UpdateStatus(taskID uint, status string) error {
    return database.DB.Model(&models.Task{}).Where("id = ?", taskID).
        Update("status", status).Error
}

func (r *TaskRepo) Delete(taskID uint) error {
    var task models.Task
    if err := database.DB.First(&task, taskID).Error; err != nil {
        return err
    }
    if task.Status == "in_progress" {
        return errors.New("cannot delete task in progress")
    }
    return database.DB.Select("Tags").Delete(&task).Error
}
```

## 5.4 可运行的 main.go

```go
package main

import (
    "fmt"
    "log"
    "taskflow/database"
    "taskflow/models"
    "taskflow/repository"
)

func main() {
    if err := database.Init(); err != nil {
        log.Fatal(err)
    }

    // 迁移
    database.DB.AutoMigrate(
        &models.User{},
        &models.UserProfile{},
        &models.Task{},
        &models.Tag{},
        &models.OperationLog{},
    )

    userRepo := &repository.UserRepo{}
    taskRepo := &repository.TaskRepo{}

    // 创建用户
    user := &models.User{
        Name:    "张三",
        Email:   "zhangsan@test.com",
        Profile: models.UserProfile{Bio: "Go 开发者"},
    }
    if err := userRepo.Create(user); err != nil {
        log.Fatal(err)
    }
    fmt.Printf("User created: ID=%d\n", user.ID)

    // 创建标签
    tag1 := models.Tag{Name: "紧急"}
    tag2 := models.Tag{Name: "前端"}
    database.DB.Create(&tag1)
    database.DB.Create(&tag2)

    // 创建任务
    task := &models.Task{
        Title:     "修复首页登录 bug",
        Priority:  9,
        CreatorID: user.ID,
    }
    if err := taskRepo.Create(task, []models.Tag{tag1, tag2}); err != nil {
        log.Fatal(err)
    }
    fmt.Printf("Task created: ID=%d, Urgent=%v\n", task.ID, task.Urgent)

    // 查询任务（带关联）
    found, _ := taskRepo.GetByID(task.ID)
    fmt.Printf("Task: %s, Creator: %s, Tags: %v\n",
        found.Title, found.Creator.Name, tagNames(found.Tags))

    // 列表查询
    tasks, total, _ := taskRepo.List(1, 10, "")
    fmt.Printf("Total tasks: %d\n", total)
    for _, t := range tasks {
        fmt.Printf("  - [%s] %s (priority: %d, urgent: %v)\n",
            t.Status, t.Title, t.Priority, t.Urgent)
    }

    // 更新状态
    taskRepo.UpdateStatus(task.ID, "in_progress")
    fmt.Println("Task status updated to in_progress")
}

func tagNames(tags []models.Tag) []string {
    names := make([]string, len(tags))
    for i, t := range tags {
        names[i] = t.Name
    }
    return names
}
```

## 5.5 复盘——GORM 心智模型

**1. 约定优于配置，但约定可覆盖**
表名、外键名、主键名都有默认规则。遵循约定时零配置，不遵循时用 struct tag 或接口覆盖。

**2. 链式调用 = 声明式地拼装查询**
`Where`、`Order`、`Limit` 是在累积条件，`Find`、`First` 才触发执行。理解了这个分界，就知道哪些操作可以安全组合。

**3. 零值是永远的坑**
Go 语言决定了 GORM 无法区分"未赋值"和"赋了零值"。`Updates` 用 struct 时要格外小心，养成用 `map` 或 `Select` 的习惯。

**4. 关联操作要用 Association 方法**
`Association("Tags").Append()` 比 `Save` 更精准——只动关联表，不动主表。

**5. 事务不难，用自动模式就好**
`DB.Transaction(func(tx))` 处理了提交、回滚、panic 恢复。业务代码只需在闭包里用 `tx` 代替 `DB`。

**6. Repository 层是值得的**
把 GORM 隔离在 model 和 repo 里，业务层只调用 `Create`、`GetByID`、`List` 这样的方法。

## 5.6 GORM 速查表

```
# 模型
gorm.Model                      → ID + 时间戳 + 软删除
gorm:"size:100;not null"        → 字段约束
gorm:"-"                        → 忽略字段
gorm:"uniqueIndex"              → 唯一索引
gorm:"many2many:table_name"     → 多对多

# CRUD
DB.Create(&obj)                 → 创建
DB.First(&obj, id)              → 单条查询
DB.Find(&list)                  → 多条查询
DB.Model(&obj).Update("k", "v") → 单字段更新
DB.Model(&obj).Updates(map)     → 多字段更新
DB.Delete(&obj, id)             → 删除（软删除）

# 查询构建
DB.Where / Or / Not             → 条件
DB.Select / Order / Limit       → 投影排序分页
DB.Group / Having               → 聚合
DB.Scopes(fn1, fn2)             → 可复用条件

# 关联
DB.Preload("Field")             → 预加载
DB.Joins("Field")               → JOIN 查询
DB.Model(&obj).Association("F") → 关联操作

# 事务
DB.Transaction(func(tx) {...})  → 自动事务
tx.Transaction(func(tx2) {...}) → 嵌套 SavePoint

# 调试
DB.Debug()                      → 打印 SQL
gorm.ErrRecordNotFound          → 记录不存在
```
